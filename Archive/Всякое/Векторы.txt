#include <iostream>
#include <vector>
#include <numeric>
using namespace std;


void main()
{
	setlocale(LC_ALL, "RUS");
//  _______________________________________________________________________________________________________________________
//                                                      Объявление вектора
//  _______________________________________________________________________________________________________________________

	vector<int> arr;   //Вектор - это объект класса                                               


//  _______________________________________________________________________________________________________________________
//                                 Инициализация элементов вектора push_back(n) и .at(i) и arr[i] 
//  _______________________________________________________________________________________________________________________

	arr.push_back(2);  //.push_back(n) - добавить число n в конец вектора
	arr.push_back(44);
	arr.push_back(77);
	arr.push_back(245);

//        ИЛИ
	arr.at(0) = 2;    // .at(i) - немного медленне чем arr[i] но надежнее
	arr.at(1) = 44;
	arr.at(2) = 77;
	arr.at(3) = 245;
//  arr.at(5) = 54;   // вызовет обибку т.к. в данный момент размер вектора 4 и не существут элемента 6. 
				      // Добавление нового элемента с увеличением массива через - .push_back(n)

//        ИЛИ
// Синтаксис как с массивом. 
	arr[0] = 2;       // arr[i] - есть риск выхода за пределы массива! (arr[i + 1])
	arr[1] = 44;
	arr[2] = 77;
	arr[3] = 245;
//  arr[5] = 54;      // вызовет обибку т.к. в данный момент размер вектора 4 и не существут элемента 6. 
				      // Добавление нового элемента с увеличением массива через - .push_back(n)

	cout << "Инициализирован массив: " << endl;
	for (int i = 0; i < arr.size(); i++)
	{
		cout << "Элемент вектора №" << i << " равен: " << arr[i] << endl;
	}
	cout << endl;

//  _______________________________________________________________________________________________________________________
//                                       Удаление последненго элемента вектора .pop_back();
//  _______________________________________________________________________________________________________________________	
	
	arr.pop_back();
	cout << "Удаление последнего элемента: " << endl;
	for (int i = 0; i < arr.size(); i++)
	{
		cout << "Элемент вектора №" << i << " равен: " << arr[i] << endl;
	}
	cout << endl;


//  _______________________________________________________________________________________________________________________
//                                                Запрос размера вектора .size()
//  _______________________________________________________________________________________________________________________
	
	cout << "Размер вектора: " << arr.size() << endl;     // Покажет 2     

//  _______________________________________________________________________________________________________________________
//                                              Запрос capacity вектора .capacity()
//  _______________________________________________________________________________________________________________________

	cout << "capacity вектора: " << arr.capacity() << endl;     // Покажет 2     


//  _______________________________________________________________________________________________________________________
//                                 Запрос accumulate вектора accumulate(arr.cbegin(), arr.cend(), 0) 
//                                                       #include <numeric>
//  _______________________________________________________________________________________________________________________

    #include <numeric>
	cout << "accumulate вектора: " << accumulate(arr.cbegin(), arr.cend(), 0) << endl;


//  _______________________________________________________________________________________________________________________
//                                               Очистка элементов вектора .clear();
//  _______________________________________________________________________________________________________________________	
	
	arr.clear();

	cout << "Размер вектора после очистки: " << arr.size() << endl;  // Покажет 0
	cout << endl;
}
